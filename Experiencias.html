#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#ifdef _WIN32
#define CLEAR "cls"
#else
#define CLEAR "clear"
#endif

void jogoModo1() {
int min, max, numeroEscolhido, numeroUsuario;
char continuar;

srand(time(0));

do {
printf("Digite o valor mínimo da faixa: ");
scanf("%d", &min);
printf("Digite o valor máximo da faixa: ");
scanf("%d", &max);

numeroEscolhido = (rand() % (max - min + 1)) + min;

printf("Tente adivinhar o número entre %d e %d: ", min, max);
scanf("%d", &numeroUsuario);

if (numeroUsuario == numeroEscolhido) {
printf("Parabéns! Você acertou!\n");
} else {
printf("Que pena! O número correto era %d.\n", numeroEscolhido);
}

printf("Deseja jogar novamente? (s/n): ");
scanf(" %c", &continuar);

system(CLEAR);
} while (continuar == 's' || continuar == 'S');
}

void jogoModo2() {
int numeroEscolhido, numeroUsuario;
float dinheiro = 100.00;

srand(time(0));

while (dinheiro >= 10.00) {
numeroEscolhido = (rand() % 50) + 1;

printf("Você tem %.2f reais.\n", dinheiro);
printf("Escolha um número entre 1 e 50: ");
scanf("%d", &numeroUsuario);

dinheiro -= 10.00;

if (numeroUsuario == numeroEscolhido) {
printf("Parabéns! Você acertou!\n");
dinheiro += 15.00;
} else {
printf("Que pena! O número correto era %d.\n", numeroEscolhido);
}

if (dinheiro < 10.00) {
printf("Você ficou sem dinheiro suficiente para continuar.\n");
}

printf("Pressione Enter para continuar...\n");
getchar();
getchar();

system(CLEAR);
}

printf("Fim de jogo. Você terminou com %.2f reais.\n", dinheiro);
}

int main() {
int escolha;

printf("Escolha o modo de jogo:\n");
printf("1. Adivinhar Número\n");
printf("2. CASSINO SIMULATOR \n");
scanf("%d", &escolha);

switch (escolha) {
case 1:
jogoModo1();
break;
case 2:
jogoModo2();
break;
default:
printf("Opção inválida!\n");
break;

}

#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define NUMEROS 6
#define LIMITE 50

#ifdef _WIN32
#define CLEAR "cls"
#else
#define CLEAR "clear"
#endif

void lerNumerosUsuario(int numerosUsuario[]) {
printf("Escolha %d números entre 1 e %d:\n", NUMEROS, LIMITE);
for (int i = 0; i < NUMEROS; i++) {
printf("Número %d: ", i + 1);
scanf("%d", &numerosUsuario[i]);
}
}

void gerarNumerosAleatorios(int numerosSorteados[]) {
for (int i = 0; i < NUMEROS; i++) {
numerosSorteados[i] = (rand() % LIMITE) + 1;
}
}

int verificarAcertos(int numerosUsuario[], int numerosSorteados[]) {
int acertos = 0;
for (int i = 0; i < NUMEROS; i++) {
for (int j = 0; j < NUMEROS; j++) {
if (numerosUsuario[i] == numerosSorteados[j]) {
acertos++;
break;
}
}
}
return acertos;
}

void exibirNumeros(const char *descricao, int numeros[]) {
printf("%s: ", descricao);
for (int i = 0; i < NUMEROS; i++) {
printf("%d ", numeros[i]);
}
printf("\n");
}

int main() {
int numerosUsuario[NUMEROS];
int numerosSorteados[NUMEROS];
int acertos;
char jogarNovamente;

srand(time(0));

do {
lerNumerosUsuario(numerosUsuario);
gerarNumerosAleatorios(numerosSorteados);

exibirNumeros("Seus números", numerosUsuario);
exibirNumeros("Números sorteados", numerosSorteados);

acertos = verificarAcertos(numerosUsuario, numerosSorteados);

if (acertos == NUMEROS) {
printf("Parabéns! Você acertou todos os números!\n");
} else {
printf("Você acertou %d número(s). Tente novamente!\n", acertos);
}

printf("Deseja jogar novamente? (s/n): ");
scanf(" %c", &jogarNovamente);

system(CLEAR);
} while (jogarNovamente == 's' || jogarNovamente == 'S');

return 0;
}
